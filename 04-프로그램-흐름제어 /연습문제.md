## 문제2
```
let x = 0; //이진수
let y = 1; //2개표현

while (x < 10) { //10개의 이진수는 총 몇개의 정보표현?
  x++; // 이진수 하난씩 늘어날때
  y *= 2; // 2배씩 표현할수있다

  console.log(`이진수 ${x} 는 ${y} 개의 정보를 표시 가능`);
}
```
